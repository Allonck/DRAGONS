[docs]def _find_edges(ad, function='polynomial', order=2, trace_threshold=1.):
    """ 
      Input:

      :param ad: AstroData object.
      :param threshold: threshold in units of sigma to be applied in
                        trace_edges()

      Output: List of Edge objects where each object contains: 
        - trace:  list of tuples (x,y) with array of edge coordinates
        - function    : name of the function being fit to the trace
        - coefficients: array of coefficients
        - order       : order of the polynomial
        - evalfunction: function evaluator
        - xlim        : (xmin,xmax) limits
        - ylim        : (ymin,ymax) limits
    """

    tc = TraceConstraints(ad) 

    # Prefilter image
    data = _prefilter(ad)

    filtered_image = enhance_edges(data,filter=tc.filter,orientation=tc.orientation)
    
    # Get a list of array of tuples (x,y) for each slit edge.
    edges = trace_edges(filtered_image, tc, trace_threshold)

    # Fit the edges and attach returning list to edge objects
    _fit_edges(edges)

    return edges

[docs]class Slit(object):
    """
     Generates a list of slit objects, each containing a pair of edge traces.
     ::

     Slit Attributues:

      id            integer reference number for slit
      region        section of the image where the slit solution is valid,
                    (x1, x2, y1, y2)   
      edges         tuple of Edge object (edge_1,edge_2) defining the 
                    long edges of the slit.
      width:        average width of the slit (calculated with the edges)

    """

    def __init__(self, id, edge_1,edge_2):
    
        self.id = id
        self.edges = (edge_1,edge_2)
        x1 = np.min(edge_1.xlim+edge_2.xlim)
        x2 = np.max(edge_1.xlim+edge_2.xlim)
        y1 = np.min(edge_1.ylim+edge_2.ylim)
        y2 = np.max(edge_1.ylim+edge_2.ylim)
        self.region = (x1, x2, y1, y2)
        self.width = edge_2.evalfunction(y1)-edge_1.evalfunction(y1)

[docs]def _associate_edges(edges, ad):
    """
       Associate the edges list with actual slits discribe
       in the MDF extension of ad.

       Some edges are residuals from the trace_edges algorithm
       and need to be discarded assuming that the slit edges are
       long continuos components.

       Input:
         Edges: List of Edge object
         ad:    Astrodata Object
            
       Output:
         slits: List of Slit objects.

    """
    if ad.dispersion_axis() == 1:
        orientation = 0
        edges = _remove_bad_x_traces(edges,ad)
    else:
        orientation = 90
        edges = _remove_bad_y_traces(edges,ad)

    slits=[]
    for sn,k in enumerate(range(0,len(edges),2)):
        slit = Slit(sn,edges[k],edges[k+1])
        slits.append(slit)
        
    return slits
        

def _remove_bad_y_traces(edges,ad):
    """ 
        Discard small edges traces remaining from the
        trace_edges algorithm. This function has a directional
        dependency in the y-axis.
        Handle orientation=90  (f2,gnirs)

        Input:
          Edges: Lists of Edge object
          ad:    AD object

        Output: 
          edges: Final list of edges that correspond to
                 the slit edges in the MDF extension.

        IMPLEMENTATION NOTES:

          - Edges are discarded if they are 200 pixels or less.

          - Edges are discarded if they start beyond the startting point of the
            rest of them.
    """

    mdf = ad['MDF'].data
    x_ccd = mdf.field('x_ccd').copy()
    y_ccd = mdf.field('y_ccd').copy()

    instrument = str(ad.instrument()).lower()
    if instrument in ['f2','flam']:
        # SOMEHOW there are not enough slits in the x_ccd's array.
        # add 4 more.
        x_ccd = np.asarray(list(x_ccd) + list(x_ccd[-1]+[33.,66.,99.,132.]))

    nslit_mdf = len(x_ccd)   # this is the number of slits in the MDF
    if nslit_mdf>len(edges):
        print "Only",len(edges),"edges out of ",nslit_mdf,"slits from MDF."


    # All edges must start near y1+sigma
    ymin = [ed.ylim[0] for ed in edges]
    ystart = np.median(ymin) + np.std(ymin)
    
    g=[]
    for k,ed in enumerate(edges):
        # edges are at least 200 pixels long in the spatial direction
        if ed.ylim[1]-ed.ylim[0] < 200:
            continue
        # If this edges start beyond ystart, skip
        if ed.ylim[0] > ystart:
            continue

        g.append(k)

    # Select no more than the number of edges in the MDF
    ns = min(nslit_mdf*2,len(g))
    edges = np.asarray(edges)[g[:ns]]
    return edges

def _remove_bad_x_traces(edges,ad):
    """
        Discard small edges traces remaining from the
        trace_edges algorithm. This function has a directional
        dependency in the x-axis.
        Handle orientation=0  (gmos)

        Input:
          Edges: Lists of Edge object
          ad:    AD object

        Output: 
          edges: Final list of edges that correspond to
                 the slit edges in the MDF extension.

        IMPLEMENTATION NOTES:

          - Edges are discarded if they are 200 pixels or less.

          - Edges are discarded if they start beyond the startting point of the
            rest of them.
    """

    mdf = ad['MDF'].data
    x_ccd = mdf.field('x_ccd')
    y_ccd = mdf.field('y_ccd')

    nslit_mdf = len(y_ccd)   # this is the number of slits in the MDF
    if nslit_mdf>len(edges):
        print "Only",len(edges),"edges out of ",nslit_mdf,"slits from MDF."

    # All edges must start near x1+sigma
    xmin = [ed.xlim[0] for ed in edges]
    xstart = np.median(xmin) + np.std(xmin)
    
    g=[]
    for k,ed in enumerate(edges):
       # edges are at least 200 pixels long in the spatial direction
       if ed.xlim[1]-ed.xlim[0] < 200:
          continue
       # If this edges start beyond xstart, skip
       if ed.xlim[0] > xstart:
          continue

       g.append(k)

    # Select no more than the number of edges in the MDF
    ns = min(nslit_mdf*2,len(g))
    edges = np.asarray(edges)[g[:ns]]
    return edges

            else: 
                pl.plot(evf1(yy1),yy1,'b',evf2(yy1),yy1,'r')

             

[docs]    def asbintable(self):
        """
            Creates a BINTABLE object from the 
            slitstrace object.
            
            Input:
               slits: list of Slit objects.

            Output:
               Returns an AD object with one BINTABLE
               extension with EXTNAME="TRACESLI" and EXTVER=1.

            **Column discription**
            
            ::

                'id'       : integer reference number for slit.
                'region'   : (x1,x2,y1,y2), window of pixel co-ords enclosing this slit.
                             The origin of these coordinates could be the lower left of the
                             original image.
                'range1'   : (x1,x2,y1,y2), range where edge_1 is valid.
                             The origin of these coordinates is the lower left of the
                             original image.
                'function1': Fit function name (default: polynomial) fitting edge_1.
                'coeff1'   : Arrray of coefficients, high to low order, such that
                             pol(x) = c1*x**2 + c2*x + c3   (for order 2).
                'order1'   : Order or polynomial (default: 2).
                'range2'   : ditto for edge_2.
                'function2': ditto for edges_2
                'coeff2'   : ditto for edges_2
                'order2'   : ditto for edges_2

                'cutrange1'   : (x1,x2,y1,y2), range where edge_1 is valid.
                                The origin of these coordinates is the lower left of the
                                cutout region.
                'cutfunction1': Fit function name (default: polynomial).
                'cutcoeff1'   : Arrray of coefficients, high to low order, such that
                                pol(x) = c1*x**2 + c2*x + c3   (for order 2)
                'cutorder1'   : Order or polynomial (default: 2).
                'cutrange2'   : ditto for edge_2
                'cutfunction2': ditto for edge_2
                'cutcoeff2'   : ditto for edge_2
                'cutorder2'   : ditto for edge_2

            TBD: Implementation::: Should 'region' have the original width?
                 Ie; NAXIS1 from the original image would be x2=NAXIS2-1 for
                 a 'gmos' image. y2=NAXIS2 for F2.
            Notes:
                The 'region' tuple encloses the slit and its cutout region size
                will be the tuple (NAXIS1,NAXIS2) when copied to an IMAGE
                extension in a FITS file. 
        """
        slits = self.slits

        # Get n_coeffs'. We are assuming they are the same for all
        # edges.
        n_coeff = len(slits[0].edges[0].coefficients)
        c1 = pf.Column (name='id',format='J')
        c2 = pf.Column (name='region',format='4E')
        c3 = pf.Column (name='range1',format='4E')
        c4 = pf.Column (name='function1',format='15A')
        c5 = pf.Column (name='order1',format='J')
        c6 = pf.Column (name='coeff1',format='%dE'%n_coeff)

        c7 = pf.Column (name='range2',format='4E')
        c8 = pf.Column (name='function2',format='15A')
        c9 = pf.Column (name='order2',format='J')
        c10 = pf.Column (name='coeff2',format='%dE'%n_coeff)

        c11 = pf.Column (name='cutrange1',format='4E')
        c12 = pf.Column (name='cutfunction1',format='15A')
        c13 = pf.Column (name='cutorder1',format='J')
        c14 = pf.Column (name='cutcoeff1',format='%dE'%n_coeff)

        c15 = pf.Column (name='cutrange2',format='4E')
        c16 = pf.Column (name='cutfunction2',format='15A')
        c17 = pf.Column (name='cutorder2',format='J')
        c18 = pf.Column (name='cutcoeff2',format='%dE'%n_coeff)

        nrows = len(slits)
        tbhdu = pf.new_table(pf.ColDefs([c1,c2,c3,c4,c5,c6,c7,c8,c9,c10,\
                            c11,c12,c13,c14,c15,c16,c17,c18]),nrows=nrows)
        tb = tbhdu    # an alias

        # Write data to table columns

        orientation = slits[0].edges[0].orientation
        for k,slit in enumerate(slits):
            edge1 = slit.edges[0]; edge2 = slit.edges[1]
            tb.data.field('id')[k]        = slit.id
            tb.data.field('region')[k]    = np.asarray(slit.region)

            # EGDE_1 DATA with respect to original image co-ords
            range1 = np.asarray(edge1.xlim+edge1.ylim)  # (x1, x2, y1, y2)
            tb.data.field('range1')[k]    = range1
            tb.data.field('function1')[k] = edge1.function
            tb.data.field('order1')[k]    = edge1.order
            tb.data.field('coeff1')[k]    = edge1.coefficients

            # EGDE_2 DATA with respect to original image co-ords
            range2 = np.asarray(edge2.xlim+edge2.ylim)  # (x1, x2, y1, y2)
            tb.data.field('range2')[k]    = range2
            tb.data.field('function2')[k] = edge2.function
            tb.data.field('order2')[k]    = edge2.order
            tb.data.field('coeff2')[k]    = edge2.coefficients

            # CUT EGDE_1 DATA with respect to cutout image co-ords
            # Applied offsets to range1 from slit.region(x1,y1) 
            xlim1 =  np.asarray(edge1.xlim)
            ylim1 =  np.asarray(edge1.ylim)
            region_x1 = slit.region[0]
            region_y1 = slit.region[2]
            tb.data.field('cutrange1')[k]    = list(xlim1-region_x1)+\
                                               list(ylim1-region_y1)
            tb.data.field('cutfunction1')[k] = edge1.function
            tb.data.field('cutorder1')[k]    = edge1.order

            # We offset the constant coefficient as well by region (x1 or y1)
            if orientation == 0: offset=region_y1
            else:                offset=region_x1
            coeff1 = edge1.coefficients.copy()
            coeff1[edge1.order] = coeff1[edge1.order] - offset
            tb.data.field('cutcoeff1')[k]    = coeff1


            # CUT EGDE_2 DATA with respect to cutout image co-ords
            # Applied offsets to range2 from slit.region(x1,y1) 
            xlim2 =  np.asarray(edge2.xlim)
            ylim2 =  np.asarray(edge2.ylim)
            tb.data.field('cutrange2')[k]    = list(xlim2-region_x1)+\
                                               list(ylim2-region_y1)
            tb.data.field('cutfunction2')[k] = edge2.function
            tb.data.field('cutorder2')[k]    = edge2.order

            # We offset the constant coefficient as well by region (x1 or y1)
            coeff2 = edge2.coefficients.copy()
            coeff2[edge2.order] = coeff2[edge2.order] - offset
            tb.data.field('cutcoeff2')[k]    = coeff2

        # Add comment to TTYPE card
        hdr = tb.header
        hdr.update('TTYPE2',hdr['TTYPE2'],
                  comment='(x1,y1,x2,y2): slit window of pixel co-ords.')
        hdr.update('TTYPE3',hdr['TTYPE3'], comment='type of fitting function.')
        hdr.update('TTYPE4',hdr['TTYPE4'], comment='Number of coefficients.')
        hdr.update('TTYPE5',hdr['TTYPE5'], 
             comment='Coeff array: c[0]*x**3 + c[1]*x**2+ c[2]*x+c[3]')
        hdr.update('TTYPE6',hdr['TTYPE6'], 
             comment='(x1,y1,x2,y2): Edge fit window definition.')
        tb.header = hdr

        # Create an AD object with this
        tabad = AstroData(tbhdu)
        tabad.rename_ext("TRACESLI", 1)

        
        return tabad
