# This module contains user level functions related to finding the slits edges
# of spectroscopic flats.

import sys
from copy import deepcopy

import numpy as np
import pyfits as pf

from astrodata import AstroData
from astrodata import Errors
from astrodata.adutils import gemLog
from gempy import geminiTools as gt

import extract as extr


[docs]def trace_slits(adinput=None, function='polynomial', order=2, trace_threshold=1.):
    """
    This user level function (ulf) is used to find the slits edges
    of spectroscopic flats, create a BINTABLE hdu with this information
    ad append it to the adoutput AstroData object.

    :param adinput: Astrodata inputs to have a BINTABLE extension added
    :type adinput: Astrodata
    :param function: Name of the fitting function to use.
    :type function: Default is 'polynomial'.
    :param order: Degree os the polynomial.
    :type order: int. default is 2
    :param trace_threshold: Threshold in units of sigma to applied to the
                            filtered image.
    :type trace_threshold: int. Default is 1.

    """

    # Instantiate the log. This needs to be done outside of the try block,
    # since the log object is used in the except block
    log = gemLog.getGeminiLog(logLevel=10)

    # The validate_input function ensures that the input is not None and
    # returns a list containing one or more AstroData objects
    adinput = gt.validate_input(adinput=adinput)

    # Define the keyword to be used for the time stamp for this user level
    # function.
    keyword = "TRACESLI"

    # Initialize the list of output AstroData objects
    adoutput_list = []

    try:
        # Loop over each input AstroData object in the input list
        for ad in adinput:

            # Check whether the my_science_step user level function has been
            # run previously
            if ad.phu_get_key_value(keyword):
                raise Errors.InputError(ad.filename+" has already been "+\
                                        "processed by 'trace_slits'.")

            # Check that is prepared flat.
            if  not "PREPARED" in ad.types:
                raise Errors.InputError(ad.filename+" has not been prepared.")
            if  ad.observation_type() != 'FLAT':
                raise Errors.InputError(ad.filename+" is not FLAT.")

            # Make a copy

            nad = deepcopy(ad)
            # Find edges in the image, return a list of Edge objects
            edges = extr._find_edges(nad, function=function, order=order,
                                        trace_threshold=trace_threshold)
            log.debug('Returning from _find_edges; returning'+str(len(edges))+' edges.')

            # Clean for bad edges in the list and cross-check with the
            # ad['MDF'] data.
            slits = extr._associate_edges(edges, nad)
            log.debug('Returning from __associate_edges; returning'+str(len(slits))+\
                            ' slits.')

            # Create a slitTrace object.
            st = extr.SlitsTrace(slits)

            # Use the slit information to prepare BINTABLE format.
            # tb_adout is of type AD
            tb_adout = st.asbintable()
            log.debug('Returning from SlitsTrace.asbintable.')
            
            # Append to the input AD object
            nad.append(tb_adout)

            # Add the appropriate time stamps to the PHU
            gt.mark_history(adinput=nad, keyword=keyword)

            # Append the output AstroData object to the list of output
            # AstroData objects
            adoutput_list.append(nad)

    except:
        # Log the message from the exception
        log.critical(repr(sys.exc_info()[1]))
        raise

    # Return the list of output AstroData objects
    return adoutput_list
